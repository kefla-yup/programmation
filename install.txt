## Node JS
# Installer NVM (gestionnaire de versions Node)
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash

# Recharger le terminal
source ~/.bashrc

# Installer Node.js
nvm install --lts

# Vérifier
node -v
npm -v
